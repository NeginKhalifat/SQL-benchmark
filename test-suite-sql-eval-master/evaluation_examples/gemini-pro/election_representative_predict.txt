SELECT r.Name, COUNT(DISTINCT e.Place) AS NumPlacesWon FROM election AS e JOIN representative AS r ON e.Representative_ID = r.Representative_ID WHERE e.Votes >= 10 GROUP BY r.Name ORDER BY e.Votes DESC
SELECT e.Place, e.Votes FROM election AS e JOIN representative AS r ON e.Representative_ID = r.Representative_ID WHERE r.State = 'California' AND e.Votes > 1000 ORDER BY e.Votes DESC LIMIT 10
SELECT COUNT(*) FROM election WHERE Representative_ID IS NULL OR Votes > 1000;
SELECT Place FROM election WHERE Vote_Percent > 50 AND Date BETWEEN '2000-01-01' AND '2020-12-31' GROUP BY Place HAVING COUNT(*) > 1
SELECT COUNT(DISTINCT Seats) AS Unique_Seats_Won, COUNT(*) AS Total_Elections_Held FROM election WHERE Date < '2020-01-01'
SELECT t1.Name FROM representative AS t1 JOIN election AS t2 ON t1.Representative_ID = t2.Representative_ID WHERE t2.Votes > 20000 AND (t2.Votes > 10000 OR t2.Vote_Percent > 0.5) GROUP BY t1.Name HAVING COUNT(DISTINCT t2.Place) > 1
SELECT election.Election_ID, election.Representative_ID, election.Date, election.Votes, election.Vote_Percent, election.Seats, election.Place FROM election JOIN representative ON election.Representative_ID = representative.Representative_ID WHERE election.Representative_ID <> 123
SELECT Name FROM representative JOIN election ON representative.Representative_ID = election.Representative_ID ORDER BY Votes DESC LIMIT 11, 10;
